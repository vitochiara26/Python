import java.util.Scanner;

public class Vacaciones{
	public static void main(String args[]){
		Scanner in = new Scanner(System.in);
		String nombre = "";
		int clave = 0, dias = 0, años = 0;
		
		System.out.println("Programa para calcular los dias de Vacaciones de la empresa Coca-cola, C.A.");
		System.out.println("Dependiendo del departamento al que pertenece y los años de servicio.");

		System.out.println("Indica el nombre del empleado.");
		nombre = in.nextLine();

		System.out.println("Indica a cual departamento pertenece " + nombre);
		System.out.println("Intoduce la clave. 1 = Atencion al cliente. 2 = Logistica. 3 = Gerencia.");
		clave = in.nextInt();

		System.out.println("Indica cuantos años de servicio tiene en la empresa " + nombre);
 		años = in.nextInt();
		
		if (clave == 1){
			if(años == 1){
				dias = 6;
			}
			else if(años >= 2 && años <= 6){
				dias = 14;
			}
			else if(años >= 7){
				dias = 20;
			}
			else{
				dias = 0;
			}
			System.out.println(nombre + ", del departamento de atencion al cliente, gozará de " + dias + " de vacaciones debido a su/s " + años + " año/s de servicio/s en la empresa");
		}
		else if (clave == 2){
 			if(años == 1){
 				dias = 7;
 			}
 			else if(años >= 2 && años <= 6){
 				dias = 15;
 			}
 			else if(años >= 7){
 				dias = 22;
 			}
 			else{
 				dias = 0;
 			}
 			System.out.println(nombre + ", del departamento de logistica, gozará de " + dias + " de vacaciones debido a su/s " + años + " año/s de servicio/s en la empresa");
 		}
		else if (clave == 3){
 			if(años == 1){
 				dias = 10;
 			}
 			else if(años >= 2 && años <= 6){
 				dias = 20;
 			}
 			else if(años >= 7){
 				dias = 30;
 			}
 			else{
 				dias = 0;
 			}
 			System.out.println(nombre + ", del departamento de gerencia, gozará de " + dias + " de vacaciones debido a su/s " + años + " año/s de servicio/s en la empresa");
 		}
		else {
			System.out.println("La clave no existe.");
		}	

 }
}
